    <script>
        async function getWeather() {
            const city = document.getElementById('city').value;
            const apiKey = '9f9552013d5bf1dd1dc67e74c08d4ca2'; // کلید API
            const url = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}&units=metric〈=fa`;

            try {
                const response = await fetch(url);
                if (!response.ok) {
                    throw new Error('شهر پیدا نشد');
                }
                const data = await response.json();
                const weatherInfo = `
                    <h2>${data.name}</h2>
                    <p>دمای کنونی: ${data.main.temp} °C</p>
                    <p>وضعیت: ${data.weather[0].description}</p>
                `;
                document.getElementById('weather').innerHTML = weatherInfo;
            } catch (error) {
                document.getElementById('weather').innerHTML = `<p>${error.message}</p>`;
            }
        }

        function updateDateTime() {
            const now = new Date();
            const options = { year: 'numeric', month: 'long', day: 'numeric', hour: '2-digit', minute: '2-digit', second: '2-digit', hour12: false };
            const dateTimeString = now.toLocaleDateString('fa-IR', options).replace('،', ' ');
            document.getElementById('date-time').innerText = `تاریخ و ساعت: ${dateTimeString}`;
        }

        // زمان و تاریخ را هر بار که صفحه باز می‌شود بروزرسانی کن
        updateDateTime();
